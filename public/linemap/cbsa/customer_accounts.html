321@app.post("/api/accounts/{number}/transfer")
322def transfer(number: str, body: models.TransferRequest):
323 # 元 XFRFUN。2 口座の残高更新 + PROCTRAN を 1 トランザクション（type=TFR）。
324 # ★実用上の意図的追加（いずれも原典機能の欠落ではない）：
325 # (1) 当座貸越限度を超える送金を拒否＝残高保護。
326 # (2) PROCTRAN を送金元(原典どおり)＋受取側にも 1 行ずつ＝各口座の明細に振替が載る。
327 # 理由は CBSA_MAPPING_AND_DIFFERENCES.md に明記。
328 if body.amount <= 0:
329 raise unprocessable("Transfer amount must be positive", fail_code="4")
330 # 同一口座への自己振替は拒否（元 XFRFUN: COMM-FACCNO=COMM-TACCNO → ABEND 'SAME'）。
331 if str(number).zfill(8) == str(body.targetAccount).zfill(8):
332 raise unprocessable("Cannot transfer to the same account", fail_code="SAME")
333 with db_cursor() as cur:
334 src = cur.execute(
335 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, number)
336 ).fetchone()
337 if src is None:
338 raise not_found(f"Source account {number} not found", fail_code="1")
339 dst = cur.execute(
340 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, body.targetAccount)
341 ).fetchone()
342 if dst is None:
343 raise not_found(f"Target account {body.targetAccount} not found", fail_code="2")
344 if src["available_balance"] - body.amount < -src["overdraft_limit"]: # 意図的追加(残高保護)
345 raise conflict("Insufficient funds (overdraft limit exceeded)", fail_code="3")
346 cur.execute(
347 "UPDATE accounts SET actual_balance=actual_balance-?, available_balance=available_balance-? "
348 "WHERE sortcode=? AND number=?",
349 (body.amount, body.amount, SORTCODE, number),
350 )
351 cur.execute(
352 "UPDATE accounts SET actual_balance=actual_balance+?, available_balance=available_balance+? "
353 "WHERE sortcode=? AND number=?",
354 (body.amount, body.amount, SORTCODE, body.targetAccount),
355 )
356 # 送金元行（原典 XFRFUN どおり：金額正・相手先を desc に）
357 _log_proctran(cur, number, PROC_TYPE["TRANSFER"], body.amount,
358 f"TRANSFER{SORTCODE}{body.targetAccount}")
359 # 受取側行（意図的追加：受取口座の明細にも振替を残す）
360 _log_proctran(cur, body.targetAccount, PROC_TYPE["TRANSFER"], body.amount,
361 f"TRANSFER{SORTCODE}{number}")
