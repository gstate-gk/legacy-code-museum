10 ******************************************************************
11 * This program gets called when someone initiates a transfer
12 * of funds (perhaps) over the counter or via a web application.
13 *
14 * This program takes an incoming sortcode/account number (key)
15 * from the account where the money is transfered from and a
16 * sortcode/acount number (key) combination for the account where
17 * the money is being transferred to. Additionally, there will
18 * also be an incoming amount.
19 *
20 * This program needs to be able to cope with the data in any
21 * format. No checking is made on overdraft limits.
22 *
23 * If the transfer can be made successfully after looking up
24 * the account information then the account information will be
25 * updated and the PROCTRAN (Processed Transaction) datastore
26 * will be updated to record the successful transaction.
27 *
28 * If the transaction is a success, then the available balance
29 * and the actual balance for the account being debited and the
30 * account being credited will be returned.
31 *
32 * If the update to either the ACCOUNT or PROCTRAN datastore is
33 * unsuccessful, then the transaction is deemed to have failed.
34 *
35 * Should there be some kind of datastore failure or other
36 * unpredictable failure we should report it, back out any
37 * updates already made and abend the transaction.
38 *
39 *****************************************************************

322def transfer(number: str, body: models.TransferRequest):
323 # 元 XFRFUN。2 口座の残高更新 + PROCTRAN を 1 トランザクション（type=TFR）。
324 # ★実用上の意図的追加（いずれも原典機能の欠落ではない）：
325 # (1) 当座貸越限度を超える送金を拒否＝残高保護。
326 # (2) PROCTRAN を送金元(原典どおり)＋受取側にも 1 行ずつ＝各口座の明細に振替が載る。
327 # 理由は CBSA_MAPPING_AND_DIFFERENCES.md に明記。
328 if body.amount <= 0:
329 raise unprocessable("Transfer amount must be positive", fail_code="4")
330 # 同一口座への自己振替は拒否（元 XFRFUN: COMM-FACCNO=COMM-TACCNO → ABEND 'SAME'）。
331 if str(number).zfill(8) == str(body.targetAccount).zfill(8):
332 raise unprocessable("Cannot transfer to the same account", fail_code="SAME")
333 with db_cursor() as cur:
334 src = cur.execute(
335 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, number)
336 ).fetchone()
337 if src is None:
338 raise not_found(f"Source account {number} not found", fail_code="1")
339 dst = cur.execute(
340 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, body.targetAccount)
341 ).fetchone()
342 if dst is None:
343 raise not_found(f"Target account {body.targetAccount} not found", fail_code="2")
344 if src["available_balance"] - body.amount < -src["overdraft_limit"]: # 意図的追加(残高保護)
345 raise conflict("Insufficient funds (overdraft limit exceeded)", fail_code="3")
346 cur.execute(
347 "UPDATE accounts SET actual_balance=actual_balance-?, available_balance=available_balance-? "
348 "WHERE sortcode=? AND number=?",

350 )

355 )

360 _log_proctran(cur, body.targetAccount, PROC_TYPE["TRANSFER"], body.amount,
361 f"TRANSFER{SORTCODE}{number}")
