8 ******************************************************************
9 * This program takes an incoming account number it then accesses
10 * the account datastore and retrieves the associated account
11 * record matching on the customer number & the account_type
12 * and then deletes it.
13 *
14 * If no matching customer_number and account_type combination can
15 * found it will return an error flag and the visual1sation layer
16 * should process this accordingly.
17 *
18 * Should there be any issues, the program will abend.
19 *
20 * It is assumed that the incoming customer number will be valid.

293@app.post("/api/accounts/{number}/debit-credit", response_model=models.AccountOut)
294def debit_credit(number: str, body: models.DebitCreditRequest):
295 # 元 DBCRFUN（緑画面 BNK1CRA = 窓口チャネル）。type=DEB/CRE は原典どおり。
296 # ★実用上の意図的追加（原典の窓口にはチェック無し）：当座貸越限度を超える出金を拒否。
297 # 理由＝無制限のマイナス残高を防ぐ残高保護。原典機能の欠落ではなく安全側の機能追加。
298 # 詳細は CBSA_MAPPING_AND_DIFFERENCES.md に明記。
299 with db_cursor() as cur:
300 row = cur.execute(
301 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, number)
302 ).fetchone()
303 if row is None:
304 raise not_found(f"Account {number} not found")
305 new_actual = row["actual_balance"] + body.amount
306 new_avail = row["available_balance"] + body.amount
307 if new_avail < -row["overdraft_limit"]: # 意図的追加（残高保護）
308 raise conflict("Insufficient funds (overdraft limit exceeded)", fail_code="3")
309 cur.execute(
310 "UPDATE accounts SET actual_balance=?, available_balance=? WHERE sortcode=? AND number=?",
311 (new_actual, new_avail, SORTCODE, number),
312 )
313 ttype = PROC_TYPE["CREDIT"] if body.amount >= 0 else PROC_TYPE["DEBIT"]
314 _log_proctran(cur, number, ttype, body.amount, "")
315 row = cur.execute(
316 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, number)
317 ).fetchone()
318 return _row_to_account(row)

321@app.post("/api/accounts/{number}/transfer")
322def transfer(number: str, body: models.TransferRequest):
323 # 元 XFRFUN。2 口座の残高更新 + PROCTRAN を 1 トランザクション（type=TFR）。
324 # ★実用上の意図的追加（いずれも原典機能の欠落ではない）：
325 # (1) 当座貸越限度を超える送金を拒否＝残高保護。
326 # (2) PROCTRAN を送金元(原典どおり)＋受取側にも 1 行ずつ＝各口座の明細に振替が載る。
327 # 理由は CBSA_MAPPING_AND_DIFFERENCES.md に明記。
328 if body.amount <= 0:
329 raise unprocessable("Transfer amount must be positive", fail_code="4")
330 # 同一口座への自己振替は拒否（元 XFRFUN: COMM-FACCNO=COMM-TACCNO → ABEND 'SAME'）。
331 if str(number).zfill(8) == str(body.targetAccount).zfill(8):
332 raise unprocessable("Cannot transfer to the same account", fail_code="SAME")
333 with db_cursor() as cur:
334 src = cur.execute(
335 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, number)
336 ).fetchone()
337 if src is None:
338 raise not_found(f"Source account {number} not found", fail_code="1")
339 dst = cur.execute(
340 "SELECT * FROM accounts WHERE sortcode=? AND number=?", (SORTCODE, body.targetAccount)
341 ).fetchone()
342 if dst is None:
343 raise not_found(f"Target account {body.targetAccount} not found", fail_code="2")
344 if src["available_balance"] - body.amount < -src["overdraft_limit"]: # 意図的追加(残高保護)
345 raise conflict("Insufficient funds (overdraft limit exceeded)", fail_code="3")
346 cur.execute(
347 "UPDATE accounts SET actual_balance=actual_balance-?, available_balance=available_balance-? "
348 "WHERE sortcode=? AND number=?",
349 (body.amount, body.amount, SORTCODE, number),
350 )
351 cur.execute(
352 "UPDATE accounts SET actual_balance=actual_balance+?, available_balance=available_balance+? "
353 "WHERE sortcode=? AND number=?",
354 (body.amount, body.amount, SORTCODE, body.targetAccount),
355 )
356 # 送金元行（原典 XFRFUN どおり：金額正・相手先を desc に）
357 _log_proctran(cur, number, PROC_TYPE["TRANSFER"], body.amount,
358 f"TRANSFER{SORTCODE}{body.targetAccount}")
359 # 受取側行（意図的追加：受取口座の明細にも振替を残す）
360 _log_proctran(cur, body.targetAccount, PROC_TYPE["TRANSFER"], body.amount,
361 f"TRANSFER{SORTCODE}{number}")
362 return {"success": "Y", "from": number, "to": body.targetAccount, "amount": body.amount}
