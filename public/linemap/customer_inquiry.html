187@app.post("/api/customers", response_model=CustomerOut, status_code=201)
188def add_customer(c: CustomerCreate):
189 """Customer Add — request id 01ACUS (LGACUS01 -> LGACDB01). The new
190 customer number is generated by the DB (was a named-counter / DB2 IDENTITY).
191 
192 The customer row and its customer_secure row are inserted in one transaction:
193 if the second insert fails the first is rolled back (the original LGACDB01/02
194 could leave a CUSTOMER row without its CUSTOMER_SECURE — an intentional
195 improvement, see GENAPP_BRANCH_COVERAGE.md G8)."""
196 with db_tx() as db:
197 cur = db.execute(
198 "INSERT INTO customer (firstName, lastName, dateOfBirth, houseName, "
199 "houseNumber, postcode, phoneHome, phoneMobile, emailAddress) "
200 "VALUES (?,?,?,?,?,?,?,?,?)",
201 (c.firstName, c.lastName, c.dateOfBirth, c.houseName, c.houseNumber,
202 c.postcode, c.phoneHome, c.phoneMobile, c.emailAddress))
203 num = cur.lastrowid
204 # The original (LGACDB02) seeds a shared default MD5 hash for every
205 # customer — a known-credential anti-pattern we do NOT carry over. New
206 # customers start with no usable password (state 'R' = reset required)
207 # and must register one via POST /api/customers/{num}/password.
208 db.execute("INSERT INTO customer_secure (customerNumber, customerPass, "
209 "state_indicator, pass_changes) VALUES (?, NULL, 'R', 0)", (num,))
210 out = c.model_dump()
211 out["customerNumber"] = num
212 out["numPolicies"] = 0
213 out["passwordStatus"] = "reset_required"
214 return out
