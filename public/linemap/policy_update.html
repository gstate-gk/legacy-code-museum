514@app.get("/api/policies/commercial", response_model=list[CommercialOut])
515def list_commercial(customerNumber: int | None = None, zipcode: str | None = None):
516 """Browse commercial policies by customer (LGIPDB01 03ICOM / Cust_Cursor)
517 or by postcode (05ICOM / Zip_Cursor). The original caps the customer browse
518 at 20 rows (IF ICOM-Record-Count > 20); the postcode browse has no cap — we
519 reproduce that asymmetry exactly (see GENAPP_BRANCH_COVERAGE.md G1)."""
520 where, params = ["p.policyType='C'"], []
521 if customerNumber is not None:
522 where.append("p.customerNumber=?")
523 params.append(customerNumber)
524 if zipcode is not None:
525 where.append("c.zipcode=?")
526 params.append(zipcode)
527 sql = ("SELECT p.policyNumber FROM policy p JOIN commercial c "
528 "ON p.policyNumber=c.policyNumber WHERE " + " AND ".join(where) +
529 " ORDER BY p.policyNumber")
530 if customerNumber is not None and zipcode is None:
531 sql += f" LIMIT {COMMERCIAL_BROWSE_CAP}" # 03ICOM: max 20 (original)
532 with db_tx() as db:
533 rows = db.execute(sql, params).fetchall()
534 return [get_commercial(r["policyNumber"]) for r in rows]
