563 * Save SQLCODE in message
564 MOVE SQLCODE TO EM-SQLRC

567 END-EXEC

571 END-EXEC

573 MOVE TIME1 TO EM-TIME

576 COMMAREA(ERROR-MSG)
577 LENGTH(LENGTH OF ERROR-MSG)
578 END-EXEC.

584 COMMAREA(CA-ERROR-MSG)

586 END-EXEC

588 MOVE DFHCOMMAREA(1:90) TO CA-DATA
589 EXEC CICS LINK PROGRAM('LGSTSQ')
590 COMMAREA(CA-ERROR-MSG)
591 LENGTH(LENGTH OF CA-ERROR-MSG)
592 END-EXEC
593 END-IF
