484*> Converts date in binary to UK/USA/Intl date format
485*>****************************************************
486*> Input: u-bin
487*> output: ws-date as uk/US/Inlt date format
488*> u-date & ws-Date = spaces if invalid date

499 go to zz060-Exit.

501 move ws-days to ws-swap
502 move ws-month to ws-days
503 move ws-swap to ws-month
504 go to zz060-Exit.
505*>
506*> So its International date format
507*>
508 move "ccyy/mm/dd" to ws-date. *> swap Intl to UK form
509 move u-date (7:4) to ws-Intl-Year.
510 move u-date (4:2) to ws-Intl-Month.
511 move u-date (1:2) to ws-Intl-Days.
512*>

38async def post_batch(db: AsyncSession, batch_id: int) -> dict:
39 """Post a closed batch to the ledger.
40 
41 Equivalent to running gl070 → gl071 → gl072 in sequence.
42 Returns summary of posted amounts.
43 """
44 batch = await db.get(Batch, batch_id)
45 if not batch:
46 raise PostingError("Batch not found")
47 if batch.status != BatchStatus.CLOSED.value:
48 raise PostingError("Batch must be closed before posting")
49 if batch.process_status != BatchProcessStatus.WAITING.value:
50 raise PostingError("Batch has already been processed")
51 # gl070.cbl (407): only General-Ledger batches are pre-processed/posted here
52 # (`not gl-batch` is skipped). Reject PL/SL batches reaching the GL poster.
53 if batch.ledger_type != LedgerType.GL.value:
54 raise PostingError(
55 f"Batch {batch.batch_number} is not a General Ledger batch "
56 f"(ledger_type={batch.ledger_type})"
57 )
58 
59 # Get current fiscal period
60 period_result = await db.execute(
61 select(FiscalPeriod).where(FiscalPeriod.is_current == True) # noqa: E712
62 )
63 fiscal_period = period_result.scalar_one_or_none()
64 
65 # Fetch all entries in this batch
66 entries_result = await db.execute(
67 select(JournalEntry)
68 .where(JournalEntry.batch_id == batch_id)
69 .order_by(JournalEntry.line_number)
70 )
71 entries = list(entries_result.scalars().all())
72 
73 if not entries:
74 raise PostingError("Batch has no entries")
75 
76 posted_count = 0
77 total_amount = Decimal(0)
78 errors: list[str] = []
79 
80 for entry in entries:
81 # --- gl070 equivalent: validate ---
82 dr_account = await _find_account(db, entry.debit_account, entry.debit_pc)
83 cr_account = await _find_account(db, entry.credit_account, entry.credit_pc)
84 
85 if not dr_account:
86 errors.append(f"Line {entry.line_number}: Debit account {entry.debit_account} not found")
87 continue
88 if not cr_account:
89 errors.append(f"Line {entry.line_number}: Credit account {entry.credit_account} not found")
90 continue
91 if dr_account.account_type == AccountType.HEADER.value:
92 errors.append(f"Line {entry.line_number}: Cannot post to header account {entry.debit_account}")
93 continue
94 if cr_account.account_type == AccountType.HEADER.value:
95 errors.append(f"Line {entry.line_number}: Cannot post to header account {entry.credit_account}")
96 continue
97 
98 # --- gl070/gl072 equivalent: update ledger balances (incl. VAT) ---
99 # Original gl070.cbl (444-475) explodes each posting into DR / CR / VAT
100 # lines before gl072 adds them to ledger-balance:
101 # DR line += amount (+vat if vat_side == "CR")
102 # CR line -= amount (+vat if vat_side == "DR")
103 # VAT line += vat (-vat if vat_side == "CR") [only if vat acct & amt]
104 vat = entry.vat_amount or Decimal(0)
105 vat_side = (entry.vat_side or "").strip().upper()
106 dr_account.balance += entry.amount + (vat if vat_side == "CR" else Decimal(0))
107 cr_account.balance -= entry.amount + (vat if vat_side == "DR" else Decimal(0))
108 
109 vat_account = None
110 if entry.vat_account and vat != 0:
111 vat_account = await _find_account(db, entry.vat_account, entry.vat_pc or 0)
112 if not vat_account:
113 errors.append(
114 f"Line {entry.line_number}: VAT account {entry.vat_account} not found"
115 )
116 continue
117 vat_account.balance += (-vat if vat_side == "CR" else vat)
118 
119 # F2: 残高が原典フィールド上限(±99,999,999.99)を超えるなら明示エラー
120 # （原典のサイレント桁落ちを継承しない）。
121 for acc in (dr_account, cr_account, vat_account):
122 if acc is not None and abs(acc.balance) > LEDGER_BALANCE_LIMIT:
123 errors.append(
124 f"Line {entry.line_number}: account {acc.account_number:04d} balance "
125 f"{acc.balance} exceeds ledger limit ±{LEDGER_BALANCE_LIMIT}"
126 )
127 
128 # Audit log entries
129 if fiscal_period:
130 db.add(AuditLog(
131 fiscal_period_id=fiscal_period.id,
132 batch_id=batch.id,
133 entry_id=entry.id,
134 account_id=dr_account.id,
135 amount=entry.amount,
136 side="D",
137 description=entry.description,
138 ))
139 db.add(AuditLog(
140 fiscal_period_id=fiscal_period.id,
141 batch_id=batch.id,
142 entry_id=entry.id,
143 account_id=cr_account.id,
144 amount=entry.amount,
145 side="C",
146 description=entry.description,
147 ))
148 if vat_account is not None:
149 db.add(AuditLog(
150 fiscal_period_id=fiscal_period.id,
151 batch_id=batch.id,
152 entry_id=entry.id,
153 account_id=vat_account.id,
154 amount=vat,
155 side=("C" if vat_side == "CR" else "D"),
156 description=f"VAT {entry.description}",
157 ))
158 
159 posted_count += 1
160 total_amount += entry.amount
161 
162 if errors:
163 # NOTE: the original gl070/gl072 do NOT validate accounts or reopen the
164 # batch on error (gl072 reads the ledger assuming the account exists). This
165 # account-existence check + reopen-on-error is an intentional safety
166 # addition in the conversion, not behaviour from the COBOL.
167 batch.status = BatchStatus.OPEN.value
168 batch.proofed_at = None
169 await db.commit()
170 raise PostingError(f"Posting failed with {len(errors)} error(s): " + "; ".join(errors))
171 
172 # Mark batch as processed
173 batch.process_status = BatchProcessStatus.PROCESSED.value
174 batch.posted_at = datetime.now()
175 await db.commit()
176 
177 return {
178 "batch_id": batch_id,
179 "batch_number": batch.batch_number,
180 "posted_count": posted_count,
181 "total_amount": str(total_amount),
182 }
