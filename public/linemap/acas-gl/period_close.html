1"""Period close service — business logic from gl080.cbl.
2 
3COBOL workflow:
41. Verify all batches are PROCESSED
52. Archive posting records
63. Roll balances (current → prior, current quarter update)
74. Advance cycle number
85. Delete processed posting records

10Modernization:
11- No file-based archiving (audit_log table replaces sequential archive)
12- Period history maintained in fiscal_periods table
13- Quarter balance tracking preserved from COBOL
14"""

29 pass

32async def close_period(db: AsyncSession) -> dict:
33 """Close current fiscal period — equivalent to gl080."""
34 settings = await db.get(Settings, 1)
35 if not settings:
36 raise PeriodCloseError("Settings not initialized")
37 
38 # 1. Verify all batches are processed
39 unprocessed = await db.execute(
40 select(Batch).where(Batch.process_status == BatchProcessStatus.WAITING.value)
41 )
42 waiting_batches = list(unprocessed.scalars().all())
43 if waiting_batches:
44 raise PeriodCloseError(
45 f"{len(waiting_batches)} batch(es) still waiting to be processed"
46 )
47 
48 # 2. Close current fiscal period
49 current_period = await db.execute(
50 select(FiscalPeriod).where(FiscalPeriod.is_current == True) # noqa: E712
51 )
52 period = current_period.scalar_one_or_none()
53 if period:
54 period.is_current = False
55 period.closed_at = datetime.now()
56 
57 # gl080.cbl (243-249): the END-OF-PERIOD (quarter roll) is gated. The original
58 # only performs Phase 5 when `scycle` is an exact multiple of `period` (the
59 # number of cycles per quarter): `divide scycle by period ... if scycle = y`.
60 # cycle_period 3 (monthly): roll every 3rd cycle (3,6,9,12 -> quarter end)
61 # cycle_period 13 (weekly): roll every 13th cycle (13,26,39,52)
62 # Design decision (慶太, 2026-06-27): close_period is called PER CYCLE and always
63 # advances the cycle; only the quarter roll is gated to cycle boundaries — so a
64 # mid-quarter close advances the cycle but does NOT snapshot quarters.
65 quarter = settings.current_quarter
66 cp = settings.cycle_period
67 quarter_boundary = cp > 0 and settings.current_cycle % cp == 0
68 
69 accounts: list[Account] = []
70 if quarter_boundary:
71 accounts_result = await db.execute(
72 select(Account).where(Account.account_type == AccountType.DETAIL.value)
73 )
74 accounts = list(accounts_result.scalars().all())
75 
76 # gl080.cbl (252-261): snapshot the current balance into the quarter slot, and
77 # at quarter 4 ALSO save it to ledger-last (prior_balance). The original does
78 # NOT zero the balance — GL balances carry forward (gl080 performs no P&L→
79 # capital close). Clearing balance here was an over-reach not in the original.
80 for acc in accounts:
81 match quarter:
82 case 1:
83 acc.q1_balance = acc.balance
84 case 2:
85 acc.q2_balance = acc.balance
86 case 3:
87 acc.q3_balance = acc.balance
88 case 4:
89 acc.q4_balance = acc.balance
90 acc.prior_balance = acc.balance # ledger-last (balance は据置)
91 
92 # 4. Archive batches
93 processed = await db.execute(
94 select(Batch).where(Batch.process_status == BatchProcessStatus.PROCESSED.value)
95 )
96 archived_count = 0
97 for batch in processed.scalars().all():
98 batch.process_status = BatchProcessStatus.ARCHIVED.value
99 batch.archived_at = datetime.now()
100 archived_count += 1
101 
102 # 5. Advance cycle
103 old_cycle = settings.current_cycle
104 settings.current_cycle += 1
105 # gl080.cbl (272-277): year-end cycle wraparound.
106 # cycle_period 3 (quarterly, 12 cycles/yr): cycle > 12 -> 1
107 # cycle_period 13 (weekly, 52 cycles/yr): cycle > 52 -> 1
108 if settings.cycle_period == 3 and settings.current_cycle > 12:
109 settings.current_cycle = 1
110 elif settings.cycle_period == 13 and settings.current_cycle > 52:
111 settings.current_cycle = 1
112 
113 # current-quarter only advances when the quarter actually rolled (gl080 loop-end
114 # `add 1 to current-quarter` sits inside the gated Phase 5 block).
115 if quarter_boundary:
116 if quarter == 4:
117 settings.current_quarter = 1
118 else:
119 settings.current_quarter += 1
120 
121 await db.commit()
122 
123 return {
124 "closed_cycle": old_cycle,
125 "new_cycle": settings.current_cycle,
126 "quarter_rolled": quarter_boundary,
127 "accounts_rolled": len(accounts),
128 "batches_archived": archived_count,
129 }
